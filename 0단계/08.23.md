## 배열 뒤집기 


### 문제 
```javascript
function solution(num_list) {
    var answer = [];
    return answer;
}
```

### 문제 풀이
```javascript
function solution(num_list) {
    var answer = [];
    return answer = num_list.reverse()
}


function solution(num_list) {
    var answer = [];
    for(let i = num_list.length - 1; i >= 0; i--) {
        answer.push(num_list[i]);
    }
    return answer;
}
```
