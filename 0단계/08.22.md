## 아이스 아메리카노

### 문제 
```javascript
function solution(money) {
    var answer = [];
    return answer;
}
```

### 문제 풀이
```javascript
function solution(money) {
    var answer = [];

    let americano = 5500;

    let avg = Math.floor(money/americano);
    let float = (money%americano);

    answer.push(avg, float);

    return answer
}

function solution(money) {
    return [Math.floor(money / 5500), money % 5500];
}
```
