```javascript
}
```

## 외계 행성의 나이

### 문제

```javascript
function solution(age) {
  var answer = "";
  return answer;
}
```

### 문제 풀이

```javascript
function solution(age) {
  let 알파벳 = ["a", "b", "c", "d", "e", "f", "g", "h", "i", "j"];

  let answer = "";
  age = age.toString();

  for (let i = 0; i < age.length; i++) {
    answer += 알파벳[age[i]];
  }
  return answer;
}
```
