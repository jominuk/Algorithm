## 최대값 만들기

### 문제

```jsx
function solution(numbers) {
    var answer = 0;
    return answer;
}
```

### 문제 풀이

```jsx
function solution(numbers) {
    let total = numbers.sort((a,b) => b-a)
    
    let max = total[0];
    let min = total[1];
    
    let answer = max * min;
    return answer;
}


function solution(numbers) {
    numbers.sort((a,b)=>b-a);
    return numbers[0]*numbers[1];
}

```
